import turtle

# Fenster einrichten
wn = turtle.Screen()
wn.title("SpongeBob Horror Spiel")
wn.bgcolor("black")
wn.setup(width=800, height=600)
wn.tracer(0)

# Spieler (SpongeBob) einrichten
player = turtle.Turtle()
player.shape("square")
player.color("yellow")
player.penup()
player.speed(0)

# Bewegung des Spielers
def move_left():
    x = player.xcor()
    x -= 20
    player.setx(x)

def move_right():
    x = player.xcor()
    x += 20
    player.setx(x)

# Tastatursteuerung
wn.listen()
wn.onkeypress(move_left, "Left")
wn.onkeypress(move_right, "Right")

# Geister einrichten
geister = []
for _ in range(3):
    geist = turtle.Turtle()
    geist.shape("triangle")
    geist.color("red")
    geist.penup()
    geist.speed(0)
    geist.goto(-100, 250)
    geister.append(geist)

# Haupt-Loop des Spiels
while True:
    wn.update()

    # Bewege die Geister
    for geist in geister:
        x = geist.xcor()
        x += 5
        geist.setx(x)

        # Überprüfe Kollisionen mit dem Spieler
        if player.distance(geist) < 20:
            player.color("red")
            player.write("Du wurdest gefangen!", align="center", font=("Courier", 24, "normal"))
            player.hideturtle()
            break
